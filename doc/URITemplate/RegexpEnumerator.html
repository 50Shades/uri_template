def each(str)
  return Enumerator.new(self,:each,str) unless block_given?
  rest = str
  loop do
    m = @regexp.match(rest)
    if m.nil?
      yield rest
      break
    end
    yield m.pre_match if m.pre_match.size > 0
    yield m
    if m[0].size == 0
      # obviously matches empty string, so post_match will equal rest
      # terminate or this will loop forever
      yield m.post_match
      break
    end
    rest = m.post_match
  end
  return self
end
