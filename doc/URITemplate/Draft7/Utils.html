def pair_array?(a)

def pair_array_to_hash(a)
  if a.kind_of? Array and pair_array?(a)
    return Hash[ *a.map{ |k,v| [k,pair_array_to_hash(v)] }.flatten(1) ]
  else
    return a
  end
end

def pct(s, m)
  s.to_s.encode('UTF-8').gsub(m){
    '%'+$1.unpack('H2'*$1.bytesize).join('%').upcase
  }
end
